def test_resize
  #image = Magick::Image.from_blob(@params['picture_file'].read).first
  cat = File.read("#{Dir.pwd}/test/unit/Cheetah.jpg")
  image = Magick::Image.from_blob(cat).first
  assert_equal image.rows, 225
  assert_equal image.columns, 225
  image = image.change_geometry("400X300"){|cols, rows, i|  i.resize!(cols,rows) }
  assert_equal image.rows, 400
  assert_equal image.columns, 300
end
